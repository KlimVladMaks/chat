```kotlin
package io.github.klimvladmaks.chat.data.repository

import android.util.Log
import io.github.klimvladmaks.chat.data.api.ApiService
import io.github.klimvladmaks.chat.data.storage.AuthStorage
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import io.github.klimvladmaks.chat.data.api.models.LoginRequest


class ChatRepository(
    private val apiService: ApiService,
    private val authStorage: AuthStorage
) {
    suspend fun login(username: String, password: String) = withContext(Dispatchers.IO) {
        try {
            val request = LoginRequest(username, password)
            val response = apiService.login(request)
            val token = response.body()?.string() ?: throw Exception()
            authStorage.saveTokenAndUsername(token, username)
        } catch (e: Exception) {
            throw Exception("Ошибка авторизации")
        }
    }

    suspend fun logout() = withContext(Dispatchers.IO) {
        val token = authStorage.getToken() ?: throw Exception("Токен отсутствует")
        try {
            val response = apiService.logout(token)
        } finally {
            authStorage.clear()
        }
    }

    fun getUsername() = authStorage.getUsername()
}
```

```kotlin
package io.github.klimvladmaks.chat.data.storage

import android.content.Context
import android.content.SharedPreferences
import androidx.core.content.edit

class AuthStorage(context: Context) {
    private val prefs: SharedPreferences = context.applicationContext.getSharedPreferences(
        PREFS_NAME,
        Context.MODE_PRIVATE
    )

    fun saveTokenAndUsername(token: String, username: String) {
        prefs.edit {
            putString(KEY_TOKEN, token)
            putString(KEY_USERNAME, username)
        }
    }

    fun getToken(): String? = prefs.getString(KEY_TOKEN, null)

    fun getUsername(): String? = prefs.getString(KEY_USERNAME, null)

    fun clear() {
        prefs.edit { clear() }
    }

    private companion object {
        private const val PREFS_NAME = "chat_auth"
        private const val KEY_TOKEN = "token"
        private const val KEY_USERNAME = "username"
    }
}
```

```kotlin
package io.github.klimvladmaks.chat.ui

import androidx.compose.runtime.Composable
import androidx.compose.runtime.remember
import androidx.compose.ui.Modifier
import androidx.lifecycle.viewmodel.compose.viewModel
import androidx.navigation.compose.NavHost
import androidx.navigation.compose.rememberNavController
import androidx.navigation.compose.composable
import io.github.klimvladmaks.chat.data.api.ApiService
import io.github.klimvladmaks.chat.data.repository.ChatRepository
import io.github.klimvladmaks.chat.data.storage.AuthStorage
import io.github.klimvladmaks.chat.ui.screens.ChannelsScreen
import io.github.klimvladmaks.chat.ui.screens.LoginScreen
import io.github.klimvladmaks.chat.viewmodel.ChannelsViewModel
import io.github.klimvladmaks.chat.viewmodel.LoginViewModel
import io.github.klimvladmaks.chat.viewmodel.factory.ViewModelFactory


sealed class Screen(val route: String) {
    object Login : Screen("login")
    object Channels : Screen("channels")
}


@Composable
fun AppNavigation(
    apiService: ApiService,
    authStorage: AuthStorage,
    modifier: Modifier = Modifier,
) {
    val navController = rememberNavController()
    val repository = remember(apiService, authStorage) {
        ChatRepository(apiService, authStorage)
    }
    val factory = remember(repository) {
        ViewModelFactory(repository)
    }

    val loginViewModel: LoginViewModel = viewModel(factory = factory)
    val channelsViewModel: ChannelsViewModel = viewModel(factory = factory)

    NavHost(
        navController = navController,
        startDestination = Screen.Login.route,
        modifier = modifier
    ) {
        composable(Screen.Login.route) {
            LoginScreen(
                loginViewModel,
                onLoginSuccess = {
                    navController.navigate(Screen.Channels.route) {
                        popUpTo(Screen.Login.route) { inclusive = true }
                    }
                }
            )
        }
        composable(Screen.Channels.route) {
            ChannelsScreen(
                channelsViewModel,
                onLogout = {
                    navController.navigate(Screen.Login.route) {
                        popUpTo(Screen.Channels.route) { inclusive = true }
                    }
                }
            )
        }
    }
}
```
