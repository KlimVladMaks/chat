Привет! Я сейчас разрабатываю учебное мобильное приложение для чата. Сейчас я работаю над модулем `api`, который отвечает за работу с API сервера. Я уже реализовал следующий `ApiService`:
```kotlin
package io.github.klimvladmaks.chat.data.api

import io.github.klimvladmaks.chat.data.api.dto.LoginRequest
import io.github.klimvladmaks.chat.data.api.dto.Message
import okhttp3.ResponseBody
import retrofit2.Response
import retrofit2.http.Body
import retrofit2.http.GET
import retrofit2.http.Headers
import retrofit2.http.POST
import retrofit2.http.Path
import retrofit2.http.Query


interface ApiService {

    @POST("/login")
    @Headers("Content-Type: application/json")
    suspend fun login(@Body credentials: LoginRequest): Response<String>

    @POST("/logout")
    suspend fun logout(): Response<Unit>

    @GET("/channels")
    suspend fun getChannels(): Response<List<String>>

    @GET("/channel/{channelName}")
    suspend fun getMessages(
        @Path("channelName") channelName: String,
        @Query("limit") limit: Int = 20,
        @Query("lastKnownId") lastKnownId: String? = null,
        @Query("reverse") reverse: Boolean = false
    ): Response<List<Message>>

    @POST("/messages")
    @Headers("Content-Type: application/json")
    suspend fun sendTextMessage(@Body message: Message): Response<ResponseBody>

    @GET("/img/{path}")
    suspend fun getFullImage(@Path("path") path: String): Response<ResponseBody>

    @GET("/thumb/{path}")
    suspend fun getThumbImage(@Path("path") path: String): Response<ResponseBody>

    @GET("/users")
    suspend fun getUsers(): Response<List<String>>
}
```
Однако меня беспокоит data class `Message`. Всё дело в том, что он должен позволять отправлять и принимать запросы следующего вида:
```json
{
  "id": "123", // игнорируется в запросах, обязательно будет в ответе
  "from": "my name", // обязательное, "от кого", должно совпадать с именем пользователя
  "to": "1@channel", // опциональное в запросах, по умолчанию "1@channel"
  "data": {
    "Text": {
      "text": "my message" // обязательно
    },
    // ИЛИ (исключающее)
    "Image": {
      "link": "path/to/pic.jpg" // опционально для запросов
    }
  },
  "time": "1297490" // UNIX time, опционально для запросов
}
```
То есть может быть как:
```json
    "Text": {
      "text": "my message" // обязательно
    }
```
так и:
```json
   "Image": {
      "link": "path/to/pic.jpg" // опционально для запросов
    }
```
И это нужно как-то обрабатывать. Знакомый предложил мне следующий вариант:

```kotlin
package io.github.klimvladmaks.chat.data.api.dto

import kotlinx.serialization.SerialName
import kotlinx.serialization.Serializable
import kotlinx.serialization.SerializationException
import kotlinx.serialization.builtins.serializer
import kotlinx.serialization.encoding.Encoder
import kotlinx.serialization.json.JsonContentPolymorphicSerializer
import kotlinx.serialization.json.JsonElement
import kotlinx.serialization.json.JsonObject
import kotlinx.serialization.json.buildJsonObject
import kotlinx.serialization.json.jsonObject
import kotlinx.serialization.serializer

@Serializable
sealed class MessageData {
    @Serializable
    @SerialName("Text")
    data class Text(val text: String) : MessageData()

    @Serializable
    @SerialName("Image")
    data class Image(val link: String) : MessageData()
}

object MessageDataSerializer : JsonContentPolymorphicSerializer<MessageData>(
    MessageData::class
) {
    override fun selectDeserializer(element: JsonElement) = when {
        "Text" in element.jsonObject -> serializer<MessageData.Text>()
        "Image" in element.jsonObject -> serializer<MessageData.Image>()
        else -> throw SerializationException("Unknown MessageData type: ${element.jsonObject.keys}")
    }

    override fun serialize(encoder: Encoder, value: MessageData) {
        val jsonObject = when (value) {
            is MessageData.Text -> buildJsonObject {
                put("Text", buildJsonObject {
                    put("text", value.text)
                })
            }
            is MessageData.Image -> buildJsonObject {
                put("Image", buildJsonObject {
                    put("link", value.link)
                })
            }
        }
        encoder.encodeSerializableValue(JsonObject.serializer(), jsonObject)
    }
}
```
